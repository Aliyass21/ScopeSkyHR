# CoreHR i18n — SKILL.md (Arabic/English Bilingual)

## Setup

```ts
// src/i18n/index.ts
import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import ar from '@/locales/ar.json'
import en from '@/locales/en.json'

i18n.use(initReactI18next).init({
  resources: { ar: { translation: ar }, en: { translation: en } },
  lng: 'ar',              // Arabic is default
  fallbackLng: 'en',
  interpolation: { escapeValue: false },
})

export default i18n
```

```ts
// src/hooks/useLanguage.ts
import { useTranslation } from 'react-i18next'
import { useEffect } from 'react'

export const useLanguage = () => {
  const { i18n } = useTranslation()

  const setLanguage = (lang: 'ar' | 'en') => {
    i18n.changeLanguage(lang)
    document.documentElement.lang = lang
    document.documentElement.dir = lang === 'ar' ? 'rtl' : 'ltr'
    localStorage.setItem('corehr-lang', lang)
  }

  useEffect(() => {
    const saved = localStorage.getItem('corehr-lang') as 'ar' | 'en' | null
    if (saved) setLanguage(saved)
  }, [])

  return { language: i18n.language as 'ar' | 'en', setLanguage, isRTL: i18n.language === 'ar' }
}
```

## Language Toggle Component

```tsx
// src/components/layout/LanguageToggle.tsx
import { useLanguage } from '@/hooks/useLanguage'
import { Button } from '@/components/ui/button'

export const LanguageToggle: React.FC = () => {
  const { language, setLanguage } = useLanguage()

  return (
    <Button
      variant="ghost"
      size="sm"
      onClick={() => setLanguage(language === 'ar' ? 'en' : 'ar')}
      className="font-medium text-sm gap-1.5"
    >
      <span className={language === 'ar' ? 'text-brand-primary font-bold' : 'text-muted-foreground'}>ع</span>
      <span className="text-muted-foreground">/</span>
      <span className={language === 'en' ? 'text-brand-primary font-bold' : 'text-muted-foreground'}>EN</span>
    </Button>
  )
}
```

## Employee Name Helper

```ts
// src/utils/employeeName.ts
import i18n from '@/i18n'
import type { Employee } from '@/types/employee'

export const getEmployeeName = (employee: Employee): string =>
  i18n.language === 'ar' ? employee.nameAr : employee.nameEn
```

## RTL Tailwind Patterns

```tsx
// Margin start/end (direction-agnostic)
<div className="ms-4">   {/* margin-inline-start */}
<div className="me-4">   {/* margin-inline-end */}

// Flex order flip for RTL icon+text pairs
<div className="flex items-center gap-2 rtl:flex-row-reverse">
  <Icon />
  <span>{t('label')}</span>
</div>

// Directional icons (arrows, chevrons)
<ChevronRight className="rtl:rotate-180 transition-transform" />

// Text alignment
<p className="text-start">   {/* left in LTR, right in RTL */}
<p className="text-end">     {/* right in LTR, left in RTL */}

// Rounded corners for sidebar active item
<div className="rounded-e-lg">  {/* rounded-inline-end */}
```
